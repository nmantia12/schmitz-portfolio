
function example_component_block_init() {
    $block_slug = 'example-component';
    $editor_css = '/assets/css/blocks.css';
    $style_css = '/assets/css/main.css';

    wp_register_script(
        'example-component-block-editor',
        get_template_directory_uri() . '/assets/js/' . $block_slug . '.js',
        array(
            'wp-editor',
            'wp-blocks',
            'wp-i18n',
            'wp-element',
        )
    );

    wp_register_style(
        'example-component-block-editor',
        get_template_directory_uri() . $editor_css,
        array()
    );

    wp_register_style(
        'example-component-block',
        get_template_directory_uri() . $style_css,
        array()
    );

    register_block_type( 'example-component/main', array(
        'editor_script' => 'example-component-block-editor',
        'editor_style'  => 'example-component-block-editor',
        'style'         => 'example-component-block',
    ) );
}
add_action( 'init', 'example_component_block_init' );
